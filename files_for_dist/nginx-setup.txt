server {
    listen 80;

    root /home/fedorov/Sites/ng.tis-frontned/www/dist/tis-frontend;
    index index.php index.html index.htm;

    server_name ng.tis-frontend;

    error_log /home/fedorov/Sites/logs/ng.tis-frontend.error.log error;
    access_log /home/fedorov/Sites/logs/ng.tis-frontend.access.log combined;


    location ~ /\.ht {
        deny all;
    }

    location / {
        try_files $uri $uri/ /index.php$is_args$args;
    }

    location ~ \.php$ {
        fastcgi_split_path_info ^(.+\.php)(/.+)$;
                fastcgi_pass unix:/var/run/php/php7.0-fpm.sock;
                fastcgi_read_timeout 120;
                fastcgi_index index.php;
                include fastcgi.conf;

    }

    #location /api/ {
    #	rewrite ^/api/([\w-]+)$ /api.php?controller=$1;
    #}

    location  /api/ {
        # proxy_pass http://testing.com/userhistory;
        proxy_pass /api.php
        proxy_redirect off;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

}
